   425→      for (const file of Array.from(files)) {
   426→        // Validate file type
   427→        const allowedTypes = ['image/jpeg', 'image/png', 'image/webp', 'image/jpg'];
   428→        if (!allowedTypes.includes(file.type)) {
   429→          toast({ title: 'Ошибка', description: `Файл "${file.name}" — недопустимый формат. Допускаются JPEG, PNG, WebP`, variant: 'destructive' });

   431→        }
   432→        // Validate file size (10MB)
   433→        if (file.size > 10 * 1024 * 1024) {
   434→          toast({ title: 'Ошибка', description: `Файл "${file.name}" слишком большой (макс. 10 МБ)`, variant: 'destructive' });
   435→          continue;
   436→        }
   437→
   438→        const uploadForm = new FormData();
   439→        uploadForm.append('file', file);
   440→
   441→        const res = await fetch('/api/upload', {
   442→          method: 'POST',
   443→          body: uploadForm,
   444→        });
   445→
   446→        if (res.ok) {
   447→          const data = await res.json();
   448→          setFormData(prev => ({ ...prev, images: [...prev.images, data.url] }));
   449→          toast({ title: 'Фото загружено', description: file.name });
   450→        } else {

   453→        }
   454→      }
   455→    } catch {
   456→      toast({ title: 'Ошибка соединения', description: 'Не удалось загрузить фото', variant: 'destructive' });
   457→    } finally {
   458→      setUploadingImage(false);
   459→      // Reset file input so the same file can be re-selected
   460→      if (fileInputRef.current) fileInputRef.current.value = '';
   461→    }
